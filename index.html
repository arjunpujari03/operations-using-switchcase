// User-input based arithmetic operations using switch case let num1 = parseFloat(prompt("Enter the first number:")); let num2 = parseFloat(prompt("Enter the second number:")); let operator = prompt("Enter operator (+, -, *, /, %):"); let result; switch (operator) { case "+": result = num1 + num2; alert(`Result: ${num1} + ${num2} = ${result}`); break; case "-": result = num1 - num2; alert(`Result: ${num1} - ${num2} = ${result}`); break; case "*": result = num1 * num2; alert(`Result: ${num1} * ${num2} = ${result}`); break; case "/": if (num2 !== 0) { result = num1 / num2; alert(`Result: ${num1} / ${num2} = ${result}`); } else { alert("Error: Division by zero is not allowed"); } break; case "%": result = num1 % num2; alert(`Result: ${num1} % ${num2} = ${result}`); break; default: alert("Invalid operator! Please use +, -, *, /, or %"); }
